export function ClickCounter() {
  const [clicks, setClicks] = useState(0);
  
  return (
    <div style={{ 
      padding: '1rem', 
      border: '1px solid var(--color-primary)', 
      borderRadius: '8px',
      margin: '1rem 0',
      textAlign: 'center',
      background: 'var(--color-surface)'
    }}>
      <p>A simple React component embedded in Markdown:</p>
      <div style={{ 
        fontSize: '2rem', 
        fontWeight: 'bold', 
        margin: '0.5rem 0',
        color: 'var(--color-primary)'
      }}>
        {clicks}
      </div>
      <button 
        onClick={() => setClicks(clicks + 1)}
        style={{
          padding: '0.5rem 1rem',
          background: 'var(--color-primary)',
          color: 'var(--color-background)',
          border: 'none',
          borderRadius: '4px',
          cursor: 'pointer',
          marginRight: '0.5rem'
        }}
      >
        Click me!
      </button>
      <button 
        onClick={() => setClicks(0)}
        style={{
          padding: '0.5rem 1rem',
          background: 'transparent',
          color: 'var(--color-primary)',
          border: '1px solid var(--color-primary)',
          borderRadius: '4px',
          cursor: 'pointer'
        }}
      >
        Reset
      </button>
    </div>
  );
}
