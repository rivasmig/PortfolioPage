## Overview

Before After is a Unity-based prototype of a "desktop hub" world built around the idea of **time as a mutable game mechanic**. Players navigate a surreal computer-shaped environment to uncover layers of hidden story and history.

export function NextStepsTracker() {
  const [checkedSteps, setCheckedSteps] = useState(new Set());
  
  const steps = [
    "Avatar & HUD Update: Replace placeholder player model and design context-sensitive cursor",
    "Mechanics & Optimization: Refine time-travel scripts and improve performance",
    "Level Expansion: Build six full levels based on concept zones",
    "Art Pass: Remove temporary pixel filter and apply consistent shaders"
  ];
  
  const toggleStep = (index) => {
    const newChecked = new Set(checkedSteps);
    if (newChecked.has(index)) {
      newChecked.delete(index);
    } else {
      newChecked.add(index);
    }
    setCheckedSteps(newChecked);
  };
  
  return (
    <div style={{ margin: '1rem 0' }}>
      <div style={{ marginBottom: '1rem', color: 'var(--color-primary)', fontWeight: 'bold' }}>
        üéØ Progress Tracker ({checkedSteps.size}/{steps.length} planned)
      </div>
      {steps.map((step, index) => (
        <div key={index} style={{
          display: 'flex',
          alignItems: 'flex-start',
          padding: '0.75rem',
          margin: '0.25rem 0',
          background: checkedSteps.has(index) ? 'rgba(0,255,0,0.1)' : 'var(--color-surface)',
          border: '1px solid var(--color-primary)',
          borderRadius: '4px',
          cursor: 'pointer'
        }}
        onClick={() => toggleStep(index)}
        >
          <span style={{ 
            marginRight: '0.75rem', 
            fontSize: '1.2rem',
            color: checkedSteps.has(index) ? 'green' : 'var(--color-primary)'
          }}>
            {checkedSteps.has(index) ? '‚úÖ' : '‚òê'}
          </span>
          <span style={{
